   ```typescript
   import type { SessionState } from '@/lib/ai/state-machine';
   import type { ExtractionResult } from '@/lib/ai/extraction-schema';

   export interface ChatMessage {
     id: string;
     role: 'user' | 'assistant';
     content: string;
     createdAt?: Date;
   }

   export type { SessionState, ExtractionResult };

   // Fields tracked for extraction progress pills
   export interface ExtractionField {
     key: keyof Pick<ExtractionResult, 'issueType' | 'urgency' | 'address'>;
     label: string;
     collected: boolean;
   }

   // Props interfaces for chat components
   export interface ChatContainerProps {
     messages: ChatMessage[];
     status: SessionState;
     isStreaming: boolean;
     extraction: ExtractionResult | null;
     onSendMessage: (message: string) => void;
     onEscalate: () => void;
     onConfirm: () => void;
     inputDisabled?: boolean;
   }
   ```
